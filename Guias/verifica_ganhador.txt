int main()
{
    int ganhador = 0;
    int tabuleiro[3][3] = {
        1, -1, 2,
        -1, 1, 1,
        -1, -1, 2
    };
    
    ganhador = verifica_ganhador(tabuleiro);

}

int verifica_ganhador(int matriz[3][3]){
    
    int i = 0, j = 0;
    
    for(i = 0; i < 3; i++){
        
        if(matriz[i][0] != -1 && matriz[0][i] != -1){
            
            if(matriz[i][0] == matriz[i][1] == matriz[i][2])
                return matriz[i][0];
            
            else if(matriz[0][i] == matriz[1][i] == matriz[2][i])
                return matriz[0][i];
        
        }
        
    }
    
    if(matriz[0][0] != -1 && matriz[0][2] != -1){
    
        if(matriz[0][0] == matriz[1][1] == matriz[2][2])
            return matriz[0][0];
            
        else if(matriz[0][2] == matriz[1][1] == matriz[2][0])
            return matriz[0][2];
            
    }
    
    else{
        
        return 0;
        
    }
    
}
